country_event = {
	id = tuscan_factions.14
	title = tuscan_factions.14.t
	desc = tuscan_factions.14.d
	picture = ADVISOR_eventPicture	# TODO
	is_triggered_only = no
	mean_time_to_happen = {
		months = 360
		# TODO modifiers?
	}
	trigger = {
		is_at_war = no
		has_factions = yes
		has_faction = ghibellines
		stability = -1
		has_any_disaster = no
	}
	immediate = {
		hidden_effect = {
			random_country = {
				limit = {
					government = monarchy
					is_at_war = no
					has_factions = yes
					has_faction = ghibellines
					faction_influence = {
						faction = ghibellines
						influence = 25
					}
				}
				save_event_target_as = visiting_country_dignitary
			}
		}
	}
	# organize an event
	option = {
		name = "tuscan_factions.14.a"
		add_opinion = {
			who = event_target:visiting_country_dignitary
			modifier = welcomed_a_faction_member
		}
		add_treasury = -30
		add_faction_influence = {
			faction = ghibellines
			influence = 10
		}
		add_prestige = 10
	}
	# Have a private meeting
	option = {
		name = "tuscan_factions.14.b"
		add_prestige = -15
	}
}
